// NOVA versão assíncrona (substituir checkAndUnlockMedals):
export async function checkAndUnlockMedalsAsync(
  newPoints: number,
  previousPoints: number,
  userId: string
): Promise<string[]> {
  // 1. Buscar count real do Supabase (não localStorage)
  const totalCompleted = await supabaseClient.getUserTotalCompletedCount(userId)
  const earnedBadgeIds = await supabaseClient.getUserEarnedBadgeIds(userId)

  // 2. Mesma lógica de unlock, mas com dados do Supabase
  const newMedals: string[] = []

  for (const badge of BADGES) {
    if (earnedBadgeIds.includes(badge.id)) continue // já tem
    if (badge.id.startsWith('global_')) continue // global medals são checadas separadamente

    // Verificar critério da medalha
    let unlocked = false
    if (badge.requirement && totalCompleted >= badge.requirement) {
      unlocked = true
    }
    if (badge.pointsRequired && newPoints >= badge.pointsRequired) {
      unlocked = true
    }

    if (unlocked) {
      newMedals.push(badge.id)
      // NÃO persiste aqui — ChallengesSection persiste via awardBadgeByKey()
    }
  }

  return newMedals
}

### AC2: Adicionar `getUserEarnedBadgeIds()` ao SupabaseClient

```typescript
async getUserEarnedBadgeIds(userId: string): Promise<string[]> {
  try {
    const { data, error } = await supabase
      .from('user_earned_badges')
      .select('badge_id')
      .eq('user_id', userId)
    if (error) throw error
    return (data || []).map(b => b.badge_id)
  } catch (err) {
    console.error('❌ getUserEarnedBadgeIds error:', err)
    return []
  }
}
```

## Risks & Blockers

### Risk 1: BADGES array — onde está definido? (Médio)
- **Problema:** `checkAndUnlockMedalsAsync()` precisa iterar sobre `BADGES`. Verificar se `BADGES` vem de `src/lib/badges-data-mock.ts`.
- **Mitigação:** `import { BADGES } from '../../../lib/badges-data-mock'` — este arquivo define os 43 medals.
- **Como verificar:** `grep -r "export.*BADGES" src/`

### Risk 2: `badge.requirement` vs `badge.pointsRequired` — interface inconsistente (Médio)
- **Problema:** A interface do Badge pode não ter `requirement` e `pointsRequired` como esperado.
- **Mitigação:** Verificar a interface `Badge` em `badges-data-mock.ts` antes de implementar.
- **Como verificar:** `grep -A 10 "interface Badge" src/lib/badges-data-mock.ts`

### Risk 3: Race condition — dois challenges completados rapidamente (Baixo)
- **Problema:** `getUserEarnedBadgeIds()` é chamado no início do `processCompletion()`. Se o usuário completar dois challenges quase ao mesmo tempo, ambos vão ver a mesma lista de badges earned e potencialmente desbloquear a mesma medalha duas vezes.
- **Mitigação:** `awardBadgeByKey()` deve ter um `ON CONFLICT DO NOTHING` no INSERT — verificar que isso está implementado. Se não, adicionar.
- **Impacto:** Baixo — caso de borda raramente acontece.

### Risk 4: `checkAndAwardGlobalMedals()` espera `string[]` ou `Set<string>`? (Baixo)
- **Problema:** A chamada atual passa `currentEarnedBadges` que é um `Set<string>`. Com a refatoração, `getUserEarnedBadgeIds()` retorna `string[]`.
- **Mitigação:** Verificar a assinatura de `checkAndAwardGlobalMedals()` no SupabaseClient e ajustar a chamada.
