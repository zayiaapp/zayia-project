const handleFinalizeMonth = async () => {
  if (!confirm('Finalizar mês? Isso vai salvar o ranking e criar prêmios pendentes!')) return

  // Busca top 3 usuários por pontos do mês atual
  const { data: topUsers } = await supabase
    .from('profiles')
    .select('id, full_name, points')
    .eq('role', 'user')
    .order('points', { ascending: false })
    .limit(3)

  if (!topUsers?.length) {
    alert('Nenhuma usuária encontrada para finalizar mês')
    return
  }

  const currentMonth = new Date().getMonth() + 1
  const currentYear = new Date().getFullYear()

  // Insere no monthly_rankings
  const { error } = await supabase.from('monthly_rankings').insert(
    topUsers.map((user, index) => ({
      user_id: user.id,
      month: currentMonth,
      year: currentYear,
      position: index + 1,
      points: user.points || 0,
    }))
  )

  if (error) {
    alert('Erro ao finalizar mês: ' + error.message)
    return
  }

  // Cria prize_payments para top 3
  const prizes = [500, 300, 100] // R$ por posição
  await supabase.from('prize_payments').insert(
    topUsers.map((user, index) => ({
      user_id: user.id,
      amount: prizes[index] || 0,
      status: 'pending',
      month: currentMonth,
      year: currentYear,
      position: index + 1,
    }))
  )

  alert('Mês finalizado! Prêmios criados como PENDENTES.')
}
```

async getPrizePayments(status?: 'pending' | 'paid' | 'cancelled'): Promise<PrizePayment[]> {
  let query = supabase
    .from('prize_payments')
    .select(`
      *,
      user:profiles(id, full_name, email, avatar_url)
    `)
    .order('created_at', { ascending: false })

  const { data, error } = await query
  if (error) throw error
  return data || []
}
```

    .eq('id', paymentId)
}

## Risks & Blockers

### Risk 1: `prize_payments` pode não ter coluna `paid_at` (Médio)
- **Problema:** A tabela pode não ter `paid_at`. Verificar schema antes de fazer update.
- **Mitigação:** `SELECT column_name FROM information_schema.columns WHERE table_name = 'prize_payments'` antes de implementar.
- **Solução se não tiver:** Criar migration para adicionar a coluna.

### Risk 2: `monthly_rankings` pode não ter coluna `position` (Baixo)
- **Mitigação:** Mesma verificação de schema antes de implementar.

### Risk 3: Busca com ILIKE pode ser lenta com muitos usuários (Baixo)
- **Problema:** `ILIKE '%query%'` não usa índices B-tree.
- **Mitigação:** Para a escala atual (dezenas/centenas de usuárias) é OK. Para > 10k usuárias, considerar `pg_trgm` no futuro.
